#include #include #define MAX 5 int stack[MAX], top = -1; void push(int x) { if (top == MAX - 1) printf("\nStack Overflow"); else { top++; stack[top] = x; } } int pop() { return stack[top--]; } void display() { int i; if (top == -1) printf("\nStack Empty"); else { printf("\nStack Elements:\n"); for (i = top; i >= 0; i--) printf("%d\n", stack[i]); } } int main() { int ch, val; while (1) { printf("\n1.Push\n2.Pop\n3.Display\n4.Exit"); printf("\nEnter Choice: "); scanf("%d", &ch); switch (ch) { case 1: if (top == MAX - 1) printf("\nStack Overflow"); else { printf("Enter Element: "); scanf("%d", &val); push(val); } break; case 2: if (top == -1) printf("\nStack Underflow"); else { val = pop(); printf("\nDeleted Element = %d", val); } break; case 3: display(); break; case 4: exit(0); default: printf("\nInvalid Choice"); } } return 0; }
